Calculate profit for last one year
----
select etf.symbol, etf.score, etf.price, count(*), round(avg(dividend), 3), round(sum(dividend), 3), round(sum(dividend)/etf.price*100, 2) as P, round(etf.adv / etf.price), etf.asp, etf.segment, etf.name
from etf, yahoo_dividend
where etf.symbol = yahoo_dividend.symbol and '2014-09-15' < yahoo_dividend.date
group by yahoo_dividend.symbol
order by etf.symbol;
